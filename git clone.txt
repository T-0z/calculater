nub1 = float(input("please enter the first number: "))
operator = input("please enter the operator: ")
nub2 = float(input("please enter the second number: "))

if operator == "+":
    print(nub1 + nub2)

elif operator == "-":
    print(nub1 - nub2)

elif operator == "*":
    print(nub1 * nub2)

elif operator == "/":
    print(nub1 / nub2)

else:
    print("wrong operator please try again! ")
